estate_fey_development1 = { # season 11 - no change needed
	icon = privilege_grant_new_world_charter
	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	influence = 0.05
	loyalty = 0
	is_valid = {
		has_unlocked_estate_privilege = { estate_privilege = estate_fey_development1 }
		NOT = {
		    has_unlocked_estate_privilege = { estate_privilege = estate_fey_development2 }
		}
	}
	can_select = {
		any_owned_province = {
			OR = {
				has_terrain = gladeway
				has_terrain = fey_gladeway
			}
		}
		OR = {
			primary_culture = wood_elf
			primary_culture = forest_goblin
			primary_culture = emerald_orc
			primary_culture = green_orc
			primary_culture = karakhanbari_orc
		}
	}
	on_granted = {
		custom_tooltip = estate_fey_development_gt
                hidden_effect = {
                	every_owned_province = {
				limit = {
					OR = {
						has_terrain = gladeway 
	                        		has_terrain = fey_gladeway
	                        	}
				}
				add_province_modifier = {
					name = estate_fey_reluctant_gladeway
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_fey_development_rt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = estate_fey_reluctant_gladeway
				}
				remove_province_modifier = estate_fey_reluctant_gladeway
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_fey_coexistence_gladeway
				}
				remove_province_modifier = estate_fey_coexistence_gladeway
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_fey_acceptance_gladeway
				}
				remove_province_modifier = estate_fey_acceptance_gladeway
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_fey_acceptance_forest
				}
				remove_province_modifier = estate_fey_acceptance_forest
			}
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 5
		modifier = {
			factor = 2
			num_of_owned_provinces_with = {
				      has_terrain = gladeway 
                                      has_terrain = fey_gladeway
				value = 8
			}
		}
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}
}
